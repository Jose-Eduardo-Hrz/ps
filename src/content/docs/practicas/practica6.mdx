### Código

```c
#include<stdio.h>
#include <stdlib.h>
#include<stdio.h>

void obtenerArregloNumerosPares( 
    int arregloNumeros[] , int arregloNumerosPares[] 
){
    int tamNumerosPares = 0;
    for( int i = 0 ; i < 500 ; i++ ){
        if( arregloNumeros[i] % 2 == 0 ){
            arregloNumerosPares[tamNumerosPares++] = arregloNumeros[i];
        }
    }
}

int obtenerArregloNumerosImpares( 
    int arregloNumeros[] , int arregloNumerosImpares[] 
){
    int tamNumerosImpares = 0;
    for( int i = 0 ; i < 500 ; i++ ){
        if( arregloNumeros[i] % 2 != 0 ){
            arregloNumerosImpares[tamNumerosImpares++] = arregloNumeros[i];
        }
    }
}

void generarArreglo( int arregloNumeros[] ){
    srand(time(NULL));
    for( int i = 0 ; i < 500 ; i++ ){
        arregloNumeros[i] = rand() % 1000 + 1;
    }
    return arregloNumeros;
}

void imprimirArreglo( int arreglo[] ){
    for( int i = 0 ; arreglo[i] != 0 ; i++ ){
        printf("%d " , arreglo[i]);
    }
}

int main() {

    int arregloNumeros [500];
    generarArreglo( arregloNumeros );

    int arregloNumerosPares[500] = { 0 };
    obtenerArregloNumerosPares( 
        arregloNumeros , arregloNumerosPares 
    );

    int arregloNumerosImpares[500] = { 0 };
    obtenerArregloNumerosImpares( 
        arregloNumeros , arregloNumerosImpares 
    );

    printf("Numeros Pares:\n");
    imprimirArreglo( arregloNumerosPares );

    printf("\nNumeros Impares:\n");
    imprimirArreglo( arregloNumerosImpares );

}
```

## Problema 2

### Descripción 

Generar un arreglo A de máximo 1000 elementos. El usuario determinará el tamaño del arreglo. Los valores del arreglo deberán generarse de manera aleatoria, en un rango [0..9].  El programa deberá obtener como salida un nuevo arreglo B que contenga la frecuencia de aparición de los  valores [0..9] en el arreglo A.

### Ejemplo 

| Entrada | Salida |
|------|:-------|
<br/>  | 15 <br/>Numeros Generados aleatoriamente [0-9]: <br/>2 3 1 9 8 3 4 1 9 9 3 2 6 4 7<br/><br/>Frecuencia de aparicion<br/>[0] = 0<br/>[1] = 2<br/>[2] = 2<br/>[3] = 3<br/>[4] = 2<br/>[5] = 0<br/>[6] = 1<br/>[7] = 1<br/>[8] = 1<br/>[9] = 3

### Código

```c 
#include<stdio.h>
#include <stdlib.h>
#include<stdio.h>

void generarArregloAleatoriamente( 
    int arregloAleatorio[] , int tam 
){
    printf( "\nNumeros Generados aleatoriamente [0-9]:\n" );
    for( int i = 0 ; i < tam ; i++ ){
        arregloAleatorio[i] = rand() % 10;
        printf( "%d " , arregloAleatorio[i] );
    }
}

void crearArregloConValorInicalCero( int frecuencia[] ){
    for( int i = 0 ; i < 10 ; i++ ){
        frecuencia[i] = 0;
    }
}

void obtenerFrecuencia( 
    int arregloAleatorio[] , int tam , int frecuencia[] 
){
    for( int i = 0 ; i < tam ; i++ ){
        frecuencia[ arregloAleatorio[i] ] += 1;
    }
}

void imprimirFrecuencia( int frecuencia[] ){
    printf( "\n\nFrecuencia de aparicion\n" );
    for( int i = 0 ; i < 10 ; i++ ){
        printf( "[%d] = %d\n" , i , frecuencia[i] );
    }
}

int main() {
    srand(time(NULL));
    int tam;
    scanf( "%d" , &tam );
    if( tam > 1000 ) return 0;
    int arregloAleatorio[tam];
    generarArregloAleatoriamente( arregloAleatorio , tam );
    int frecuencia[10];
    crearArregloConValorInicalCero( frecuencia );
    obtenerFrecuencia( arregloAleatorio , tam , frecuencia );
    imprimirFrecuencia( frecuencia );
}

```
